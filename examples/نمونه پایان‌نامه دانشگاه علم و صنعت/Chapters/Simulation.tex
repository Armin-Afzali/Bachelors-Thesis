\chapter{شبیه‌سازی}
\label{chap:simulation}

در این فصل، آزمایش‌های انجام‌شده برای ارزیابی روش پیشنهادی \gls{TWAC} تشریح می‌شود. ابتدا در \autoref{sec:sim_env}، محیط آزمایش و تنظیمات پیاده‌سازی بیان می‌گردد. سپس در \autoref{sec:sim_baselines}، روش‌های مقایسه‌شده معرفی می‌شوند. در \autoref{sec:sim_metrics}، معیارهای سنجش کارایی تعریف می‌گردند. در \autoref{sec:sim_experiments}، چهار آزمایش اصلی این پژوهش و نتایج آن‌ها به‌تفصیل ارائه و تحلیل می‌شوند و در نهایت در \autoref{sec:sim_summary}، جمع‌بندی نتایج ارائه می‌گردد.

\subsection{مجموعه داده}

در آزمایش‌های این پژوهش از مجموعه داده \gls{CIFAR}-10 استفاده شده است. این مجموعه داده شامل ۵۰۰۰۰ تصویر آموزشی و ۱۰۰۰۰ تصویر آزمون در ۱۰ کلاس مجزا است. هر تصویر دارای ابعاد $32 \times 32$ پیکسل با سه کانال رنگی است. این مجموعه داده به دلایل زیر انتخاب شده است:

\begin{itemize}
    \tick چالش‌برانگیز بودن برای شبکه‌های کوچک، اما قابل آموزش در زمان معقول
    \tick استاندارد بودن در ادبیات یادگیری فدرال که مقایسه با کارهای دیگر را ممکن می‌سازد
    \tick امکان ایجاد توزیع‌های \gls{NonIID} واضح با روش دیریکله
\end{itemize}

\subsection{مدل شبکه عصبی}

مدل مورد استفاده یک شبکه عصبی پیچشی ساده (\gls{CNN}) با معماری زیر است:

\begin{itemize}
    \tick \lr{Conv2d(3, 32, 3, padding=1)} $\to$ \lr{ReLU}
    \tick \lr{Conv2d(32, 64, 3, padding=1)} $\to$ \lr{ReLU} $\to$ \lr{MaxPool(2)}
    \tick \lr{Conv2d(64, 64, 3, padding=1)} $\to$ \lr{ReLU} $\to$ \lr{MaxPool(2)}
    \tick \lr{FC(4096, 128)} $\to$ \lr{ReLU}
    \tick \lr{FC(128, 10)}
\end{itemize}

این مدل دارای حدود ۵۸۲۰۰۰ پارامتر قابل‌آموزش است. انتخاب یک مدل نسبتاً کوچک به این دلیل است که هدف آزمایش‌ها مقایسه رفتار روش‌های تجمیع مختلف است، نه رسیدن به بالاترین دقت ممکن.

\subsection{توزیع داده میان گره‌ها}

برای ایجاد توزیع \gls{NonIID}، از روش دیریکله با پارامتر $\alpha_{dir}$ استفاده شده است. بیست گره داریم و مجموعه آموزشی به گونه‌ای تقسیم می‌شود که هر گره سهمی از نمونه‌های هر کلاس دریافت کند که از توزیع $\text{Dir}(\alpha_{dir})$ نمونه‌برداری شده است. در آزمایش پایه، $\alpha_{dir} = 0.5$ انتخاب شده که ناهمگونی متوسطی ایجاد می‌کند؛ در آزمایش بررسی ناهمگونی داده، مقادیر دیگری نیز بررسی می‌شوند.

\subsection{شبیه‌سازی ناهمگونی}

\subsubsection{ناهمگونی محاسباتی}

هر گره یک ضریب سرعت محاسباتی $s_i$ دارد که به‌صورت یکنواخت از بازه $[0.3, 1.0]$ نمونه‌برداری شده است. این ضریب زمان شبیه‌سازی‌شده برای آموزش محلی را تعیین می‌کند و موجب می‌شود همه گره‌ها در یک دور به‌طور هم‌زمان آموزش خود را به پایان نرسانند.

\subsubsection{ناهمگونی شبکه و گره‌های کُند}

برای آزمایش مدیریت گره‌های کُند، درصد معینی از گره‌ها به‌عنوان \gls{Straggler} تعیین می‌شوند. این گره‌ها علاوه بر سرعت محاسباتی پایین، ضریب کُند شدن شبکه‌ای نیز دارند که در شبیه‌سازی به‌صورت ضریب کُندی $3\times$ نسبت به گره‌های عادی مدل‌سازی شده است؛ به این معنا که این گره‌ها زمان بسیار بیشتری برای پاسخ‌گویی در هر دور ارتباطی نیاز دارند.


\section{روش‌های مورد مقایسه}
\label{sec:sim_baselines}

در تمام آزمایش‌ها، روش \gls{TWAC} با سه روش پایه مقایسه می‌شود:

\begin{enumerate}
    \item \textbf{\lr{FedAvg}}: الگوریتم استاندارد میانگین‌گیری فدرال بدون هیچ مکانیزم دفاعی.
    
    \item \textbf{میانگین کوتاه‌شده (\lr{TrimmedMean})}: در هر مختصه، $20\%$ از بزرگ‌ترین و کوچک‌ترین مقادیر حذف می‌شود.
    
    \item \textbf{\lr{Multi-Krum}}: در هر دور، $5$ گره با کمترین امتیاز \lr{Krum} انتخاب می‌شوند و میانگین وزن‌دار آن‌ها محاسبه می‌شود.
\end{enumerate}

هر چهار روش از یک کد پایه مشترک، یک مدل یکسان، و یک بذر تصادفی یکسان استفاده می‌کنند تا مقایسه تا حد امکان منصفانه باشد. تنها بخش متفاوت میان آن‌ها، الگوریتم تجمیع در سرور است.


\section{معیارهای سنجش کارایی}
\label{sec:sim_metrics}

\subsection{دقت آزمون نهایی و بهترین دقت}

دو معیار اصلی گزارش‌شده برای هر آزمایش، دقت آزمون در آخرین دور ($\text{Acc}^T$، که از این پس «دقت نهایی» نامیده می‌شود) و بالاترین دقت آزمون مشاهده‌شده در طول کل آموزش («بهترین دقت») هستند:

\begin{equation}
    \text{Acc}^t = \frac{1}{|\mathcal{D}_{test}|} \sum_{(x,y) \in \mathcal{D}_{test}} \mathbf{1}[\arg\max f(x; w^t) = y]
    \label{eq:accuracy}
\end{equation}

گزارش هر دو معیار به این دلیل مفید است که دقت نهایی منعکس‌کننده وضعیت مدل در پایان بودجه ارتباطی مشخص‌شده است، در حالی که بهترین دقت نشان می‌دهد روش تا چه حد توانسته در طول مسیر آموزش به یک نقطه مطلوب برسد (حتی اگر در دور آخر کمی افت داشته باشد).

\subsection{خطای آزمون نهایی}

خطای آزمون با استفاده از تابع خطای اینتروپی متقاطع محاسبه می‌شود:

\begin{equation}
    \mathcal{L}^t = -\frac{1}{|\mathcal{D}_{test}|} \sum_{(x,y) \in \mathcal{D}_{test}} \log f_y(x; w^t)
    \label{eq:test_loss}
\end{equation}

\subsection{زمان اجرا}

برای هر آزمایش، زمان کل اجرای فرایند آموزش (بر حسب ثانیه) نیز ثبت شده است. این معیار به‌طور مستقیم پیچیدگی محاسباتی عملی هر روش تجمیع را نشان می‌دهد و امکان مقایسه هزینه واقعی \gls{TWAC} با \lr{Multi-Krum} را، که از نظر نظری پیچیدگی بسیار بالاتری دارد، فراهم می‌کند.


\section{آزمایش‌ها و تحلیل نتایج}
\label{sec:sim_experiments}

چهار آزمایش اصلی این پژوهش، به ترتیب، رفتار پایه روش‌ها بدون حمله، مقاومت در برابر حمله جهت‌معکوس‌سازی در دو سطح شدت، تأثیر درجه ناهمگونی داده، و مقاومت در برابر گره‌های کُند را بررسی می‌کنند. در همه آزمایش‌ها، تنظیمات جدول \ref{tab:fl_settings} به‌عنوان پایه استفاده شده و تنها پارامتر مورد بررسی هر آزمایش تغییر کرده است.

\subsection{آزمایش ۱: مقایسه پایه (بدون حمله)}
\label{subsec:exp1}

در این آزمایش، هیچ گره مخربی وجود ندارد و هدف مقایسه رفتار پایه روش‌های مختلف در شرایط عادی محیط ابری چندگانه (با ناهمگونی محاسباتی و $\alpha_{dir}=0.5$) است.

\begin{table}[t]
    \centering
    \renewcommand{\arraystretch}{1.8}
    \caption{نتایج آزمایش پایه (بدون حمله، $\alpha_{dir} = 0.5$)}
    \label{tab:baseline_results}
    \begin{tabular}{lcccc}
        \toprule
        \textbf{روش} & \textbf{دقت نهایی} & \textbf{بهترین دقت} & \textbf{خطای نهایی} & \textbf{زمان اجرا} \\
        \midrule
        \lr{FedAvg}          & $80.12\%$ & $80.12\%$ & $0.7066$ & $2454.3$ ثانیه \\

        \lr{Multi-Krum}      & $74.94\%$ & $76.74\%$ & $1.0108$ & $2413.4$ ثانیه \\

نتایج جدول \ref{tab:baseline_results} چند نکته مهم را نشان می‌دهد. نخست، در نبود هرگونه حمله، \gls{TWAC} عملکردی تقریباً معادل \lr{FedAvg} دارد ($80.06\%$ در مقابل $80.12\%$ دقت نهایی)، با این تفاوت که خطای نهایی \gls{TWAC} ($0.6927$) اندکی بهتر از \lr{FedAvg} ($0.7066$) است. این نتیجه با انتظار طراحی روش همخوانی دارد: در شرایط عادی که همه گره‌ها صادق هستند، مکانیزم اعتماد نباید جریمه قابل‌توجهی به سیستم تحمیل کند، و نتایج تأیید می‌کنند که این هدف برآورده شده است.

دوم، \lr{Multi-Krum} حتی در این سناریوی بدون حمله، پایین‌ترین دقت ($74.94\%$) را دارد. این به دلیل محدودیت ذاتی \lr{Krum} است: این روش تنها $5$ گره از میان $10$ گره منتخب هر دور را برای تجمیع انتخاب می‌کند و از اطلاعات نیمی از گره‌های صادق صرف‌نظر می‌کند، که موجب کاهش کیفیت یادگیری حتی در نبود حمله می‌شود.

سوم، از نظر زمان اجرا، چهار روش تفاوت چشمگیری ندارند ($2411$ تا $2454$ ثانیه). این مشاهده در آزمایش حاضر، که با $20$ گره و ابعاد مدل نسبتاً کوچک ($\approx 582000$ پارامتر) انجام شده، نشان می‌دهد که در این مقیاس، هزینه غالب اجرا صرف آموزش محلی روی گره‌ها می‌شود، نه الگوریتم تجمیع در سرور. با این حال، تحلیل نظری پیچیدگی (\autoref{sec:complexity}) همچنان نشان می‌دهد که با افزایش تعداد گره‌ها یا ابعاد مدل، شکاف میان \lr{Multi-Krum} با پیچیدگی $O(n^2d)$ و \gls{TWAC} با پیچیدگی $O(nd)$ بیشتر خواهد شد.

نکته قابل‌توجه دیگر این است که در این آزمایش، با مدل پیچشی نسبتاً کوچک و $20$ گره، افت دقت \lr{FedAvg} نسبت به سناریوی بدون حمله محدود است (از $80.12\%$ در آزمایش پایه به حدود $73\%$ در حضور حمله)، که نشان می‌دهد در این مقیاس، خود فرایند میانگین‌گیری وزن‌دار بر اساس تعداد نمونه تا حدی اثر گره‌های مخرب را با اثر گره‌های صادق دیگر کم‌رنگ می‌کند؛ اما همچنان \gls{TWAC} با شناسایی فعال این گره‌ها از طریق مؤلفه همسویی جهتی، در هر دو سطح حمله برتری پایدار خود را حفظ می‌کند. همچنین مشاهده می‌شود که \lr{Multi-Krum} در سطح $20\%$ حمله ضعیف‌ترین عملکرد را دارد، در حالی که در سطح $30\%$ بهبود نسبی نشان می‌دهد؛ این رفتار با حساسیت بالای انتخاب $m$ گره ثابت در \lr{Krum} به نسبت دقیق گره‌های صادق به مخرب در هر دور قابل توضیح است.

لازم به ذکر است که زمان اجرای \gls{TWAC} در این آزمایش ($2680.3$ ثانیه) به‌طور نسبی بالاتر از سایر روش‌ها است؛ این تفاوت ناشی از سربار محاسبه میانه زمان پاسخ‌دهی برای تعیین زمان مهلت تطبیقی (معادله \eqref{eq:timeout}) و نگهداری امتیازات اعتماد در طول دورهای متعدد گره‌های کُند است، اگرچه این سربار در مقیاس بزرگ همچنان از مرتبه $O(n)$ و قابل چشم‌پوشی است.

جدول \ref{tab:overall_summary} مجموع نتایج شش سناریوی آزمایشی را در یک نگاه نمایش می‌دهد. نکات کلیدی که از مجموع این نتایج می‌توان استخراج کرد به شرح زیر است:

\begin{itemize}
    \tick \textbf{رفتار در غیاب حمله}: \gls{TWAC} در سناریوی پایه عملکردی تقریباً برابر با \lr{FedAvg} دارد ($80.06\%$ در مقابل $80.12\%$) و خطای نهایی کمی بهتر کسب می‌کند. این نتیجه نشان می‌دهد که اضافه‌کردن مکانیزم اعتماد، هزینه‌ای در شرایط عادی به سیستم تحمیل نمی‌کند.
    
    \tick \textbf{مقاومت در برابر حمله جهت‌معکوس‌سازی}: \gls{TWAC} در هر دو سطح $20\%$ و $30\%$ گره مخرب، بالاترین دقت نهایی را در میان چهار روش دارد، اگرچه فاصله آن با \lr{FedAvg} در این آزمایش‌ها محدود (کمتر از یک واحد درصد تا حدود یک واحد درصد) است.
    
    \tick \textbf{ناهمگونی داده}: در ناهمگونی شدید ($\alpha_{dir}=0.1$)، \gls{TWAC} و میانگین کوتاه‌شده عملکردی نزدیک به هم و به‌وضوح بهتر از \lr{FedAvg} و \lr{Multi-Krum} دارند. \lr{Multi-Krum} در این سناریو بیشترین آسیب را می‌بیند که نشان‌دهنده حساسیت بالای این روش به ناهمگونی شدید داده است.
    
    \tick \textbf{گره‌های کُند}: \lr{Multi-Krum} در هر دو سناریوی ناهمگونی داده و گره‌های کُند، ضعیف‌ترین عملکرد را دارد. در سناریوی گره کُند، \gls{TWAC} با کاهش تدریجی اعتماد به جای حذف کامل، نتیجه‌ای نزدیک به \lr{FedAvg} و بهتر از \lr{Multi-Krum} به دست می‌آورد.
    
    \tick \textbf{روند کلی}: در میان چهار روش، \lr{Multi-Krum} در پنج از شش سناریو ضعیف‌ترین عملکرد را دارد، که با پیچیدگی محاسباتی بالاتر آن ($O(n^2d)$) نیز همراه است. \gls{TWAC} و میانگین کوتاه‌شده در بیشتر سناریوها رقابتی نزدیک دارند، با این تفاوت که \gls{TWAC} پیچیدگی محاسباتی پایین‌تری ($O(nd)$، برابر با \lr{FedAvg}) دارد و علاوه‌بر این، تنها روشی است که به‌طور هم‌زمان از حافظه اعتماد تاریخی و مدیریت تطبیقی گره‌های کُند بهره می‌برد.
\end{itemize}

\begin{note}
    تفاوت‌های دقت گزارش‌شده در این آزمایش‌ها (معمولاً در بازه نیم تا چند واحد درصد، به‌استثنای سناریوی ناهمگونی شدید) بر اساس یک اجرای منفرد برای هر پیکربندی به‌دست آمده‌اند. تکرار آزمایش‌ها با بذرهای تصادفی متفاوت و گزارش میانگین و انحراف معیار، گام طبیعی بعدی برای افزایش استحکام آماری این نتایج است که در \autoref{sec:futurework} به آن اشاره خواهد شد.
\end{note}
